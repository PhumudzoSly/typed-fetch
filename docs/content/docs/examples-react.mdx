# React Integration

Complete examples of using typed-fetch in React applications.

## Basic Hook

Create a custom hook to fetch data:

```tsx
import { useState, useEffect } from 'react';
import { typedFetch, TypedFetchResult } from '@phumudzo/typed-fetch';

export function useFetch<K extends string>(
  url: string,
  endpointKey: K
): {
  result: TypedFetchResult<K> | null;
  loading: boolean;
  refetch: () => Promise<void>;
} {
  const [result, setResult] = useState<TypedFetchResult<K> | null>(null);
  const [loading, setLoading] = useState(true);

  const refetch = async () => {
    setLoading(true);
    const res = await typedFetch(url, { method: 'GET' }, { endpointKey });
    setResult(res);
    setLoading(false);
  };

  useEffect(() => {
    refetch();
  }, [url, endpointKey]);

  return { result, loading, refetch };
}
```

## Component with Data Display

```tsx
import { useFetch } from './hooks/useFetch';

interface User {
  id: number;
  name: string;
  email: string;
}

export function UserProfile({ userId }: { userId: number }) {
  const { result, loading } = useFetch(
    `/api/user/${userId}`,
    'GET /user/:id'
  );

  if (loading) return <div>Loading...</div>;

  if (result?.status === 200) {
    return (
      <div>
        <h1>{result.data.name}</h1>
        <p>{result.data.email}</p>
      </div>
    );
  }

  if (result?.status === 404) {
    return <div>User not found</div>;
  }

  if (result?.status === 0) {
    return <div>Network error: {result.error?.message}</div>;
  }

  return <div>Error: {result?.status}</div>;
}
```

## POST Request with Form

```tsx
import { useState } from 'react';
import { typedFetch } from '@phumudzo/typed-fetch';

export function CreateUserForm() {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [success, setSuccess] = useState(false);

  const handleSubmit = async (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    setLoading(true);
    setError(null);

    const formData = new FormData(e.currentTarget);
    const name = formData.get('name') as string;
    const email = formData.get('email') as string;

    const result = await typedFetch(
      '/api/user',
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name, email })
      },
      { endpointKey: 'POST /user' }
    );

    setLoading(false);

    if (result.status === 201) {
      setSuccess(true);
      e.currentTarget.reset();
    } else if (result.status === 400) {
      setError(result.data?.message || 'Validation error');
    } else if (result.status === 0) {
      setError(result.error?.message || 'Network error');
    } else {
      setError('Failed to create user');
    }
  };

  if (success) {
    return <div className="success">User created successfully!</div>;
  }

  return (
    <form onSubmit={handleSubmit}>
      <input name="name" type="text" placeholder="Name" required />
      <input name="email" type="email" placeholder="Email" required />
      <button disabled={loading}>
        {loading ? 'Creating...' : 'Create User'}
      </button>
      {error && <div className="error">{error}</div>}
    </form>
  );
}
```

## List with Pagination

```tsx
import { useState, useEffect } from 'react';
import { typedFetch } from '@phumudzo/typed-fetch';

interface Post {
  id: number;
  title: string;
  author: string;
}

export function PostsList() {
  const [posts, setPosts] = useState<Post[]>([]);
  const [page, setPage] = useState(1);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    const loadPosts = async () => {
      setLoading(true);
      const result = await typedFetch(
        `/api/posts?page=${page}`,
        { method: 'GET' },
        { endpointKey: 'GET /posts' }
      );

      if (result.status === 200) {
        setPosts(result.data.posts);
      }
      setLoading(false);
    };

    loadPosts();
  }, [page]);

  return (
    <div>
      <div className="posts">
        {posts.map((post) => (
          <article key={post.id}>
            <h2>{post.title}</h2>
            <p>By {post.author}</p>
          </article>
        ))}
      </div>
      <div className="pagination">
        <button
          onClick={() => setPage(p => p - 1)}
          disabled={page === 1 || loading}
        >
          Previous
        </button>
        <span>Page {page}</span>
        <button
          onClick={() => setPage(p => p + 1)}
          disabled={loading}
        >
          Next
        </button>
      </div>
    </div>
  );
}
```

## Error Boundary Integration

```tsx
import React, { ReactNode } from 'react';

interface ErrorBoundaryProps {
  children: ReactNode;
}

interface ErrorBoundaryState {
  hasError: boolean;
  error: Error | null;
}

export class ApiErrorBoundary extends React.Component<
  ErrorBoundaryProps,
  ErrorBoundaryState
> {
  constructor(props: ErrorBoundaryProps) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error };
  }

  render() {
    if (this.state.hasError) {
      return <div className="error">Something went wrong</div>;
    }

    return this.props.children;
  }
}

// Usage:
export function App() {
  return (
    <ApiErrorBoundary>
      <UserProfile userId={1} />
    </ApiErrorBoundary>
  );
}
```

## Context for API Client

```tsx
import { createContext, useContext, ReactNode } from 'react';
import { typedFetch } from '@phumudzo/typed-fetch';

interface ApiContextType {
  fetch: typeof typedFetch;
  baseUrl: string;
}

const ApiContext = createContext<ApiContextType | null>(null);

export function ApiProvider({
  children,
  baseUrl = '/api'
}: {
  children: ReactNode;
  baseUrl?: string;
}) {
  return (
    <ApiContext.Provider value={{ fetch: typedFetch, baseUrl }}>
      {children}
    </ApiContext.Provider>
  );
}

export function useApi() {
  const context = useContext(ApiContext);
  if (!context) {
    throw new Error('useApi must be used within ApiProvider');
  }
  return context;
}

// Usage:
function MyComponent() {
  const { fetch, baseUrl } = useApi();

  const loadData = async () => {
    const result = await fetch(
      `${baseUrl}/data`,
      { method: 'GET' },
      { endpointKey: 'GET /data' }
    );
  };

  return <button onClick={loadData}>Load Data</button>;
}
```

## With Loading States & Skeletons

```tsx
import { useState, useEffect } from 'react';
import { typedFetch } from '@phumudzo/typed-fetch';

export function UserCard({ userId }: { userId: number }) {
  const [result, setResult] = useState<any>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const load = async () => {
      setLoading(true);
      const res = await typedFetch(
        `/api/user/${userId}`,
        { method: 'GET' },
        { endpointKey: 'GET /user/:id' }
      );
      setResult(res);
      setLoading(false);
    };

    load();
  }, [userId]);

  if (loading) {
    return (
      <div className="card skeleton">
        <div className="skeleton-line" />
        <div className="skeleton-line" />
      </div>
    );
  }

  if (result?.status === 200) {
    return (
      <div className="card">
        <h3>{result.data.name}</h3>
        <p>{result.data.email}</p>
      </div>
    );
  }

  return <div className="card error">Failed to load</div>;
}
```
